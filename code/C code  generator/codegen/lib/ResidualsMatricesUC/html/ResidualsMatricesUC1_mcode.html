1%--------------------------------------------------------------------------
 2%--------------------------------------------------------------------------
 3% Residuals and Stiffness matrices
 4%--------------------------------------------------------------------------
 5%--------------------------------------------------------------------------
 6
 7function asmb          =  ResidualsMatricesUC(asmb,derivatives,Piola_vect,vect_kin,kinematics,IntWeight,dim,n_node_elem,ngauss)
 8
 9BFx                    =  zeros(dim^2,dim*n_node_elem);
10H                      =  zeros(dim,dim);
11QF                     =  zeros(dim^2);
12QSigmaH                =  zeros(dim^2);
13BFxT                   =  zeros(dim*n_node_elem,dim^2);
14BHx                    =  zeros(dim^2,dim*n_node_elem);
15BHxT                   =  zeros(dim*n_node_elem,dim^2);
16BJx                    =  zeros(dim^2,dim*n_node_elem);
17BJxT                   =  zeros(dim*n_node_elem,dim^2);
18KFF                    =  zeros(dim^2);
19KFH                    =  zeros(dim^2);
20KFJ                    =  zeros(dim^2,1);
21KHH                    =  zeros(dim^2);
22KHJ                    =  zeros(dim^2,1);
23KJJ                    =  0;
24geometric              =  zeros(dim*n_node_elem);
25Kxx                    =  zeros(dim*n_node_elem);
26
27for igauss=1:ngauss
28    BFx                =  vect_kin.BF(:,:,igauss);
29    H                  =  kinematics.H(:,:,igauss);
30    QF                 =  vect_kin.QF(:,:,igauss);
31    QSigmaH            =  vect_kin.QSigmaH(:,:,igauss);    
32    %----------------------------------------------------------------------
33    % Transpose of matrices
34    %----------------------------------------------------------------------
35    BFxT               =  BFx';
36    BHx                =  QF*BFx;
37    BHxT               =  BHx';
38    BJx                =  reshape(H',[],1)'*BFx;
39    BJxT               =  BJx';
40    %----------------------------------------------------------------------
41    % Stiffness matrices for Kxx
42    %----------------------------------------------------------------------
43    KFF               =  BFxT*(derivatives.D2U.D2UDFDF(:,:,igauss)*BFx);
44    KFH               =  BFxT*(derivatives.D2U.D2UDFDH(:,:,igauss)*BHx);
45    KFJ               =  BFxT*(derivatives.D2U.D2UDFDJ(:,:,igauss)*BJx);
46    KHH               =  BHxT*(derivatives.D2U.D2UDHDH(:,:,igauss)*BHx);
47    KHJ               =  BHxT*(derivatives.D2U.D2UDHDJ(:,:,igauss)*BJx);
48    KJJ               =  BJxT*(derivatives.D2U.D2UDJDJ(igauss)*BJx);
49    geometric         =  (BFxT*QSigmaH*BFx) + derivatives.DU.DUDJ(igauss)*(BFxT*BHx);
50    Kxx               =  KFF + KHH + KJJ + (KFH + KFH') + (KFJ + KFJ') + (KHJ + KHJ') + geometric;
51    %----------------------------------------------------------------------
52    % Residual and stiffness matrices for non-linear elasticity regions
53    %----------------------------------------------------------------------
54    asmb.Tx           =  asmb.Tx  +  (BFx'*Piola_vect(:,igauss))*IntWeight(igauss);
55    asmb.Kxx          =  asmb.Kxx  +  Kxx*IntWeight(igauss);
56end

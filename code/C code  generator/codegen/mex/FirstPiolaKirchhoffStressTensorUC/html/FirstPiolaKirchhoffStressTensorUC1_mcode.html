1%--------------------------------------------------------------------------
 2%--------------------------------------------------------------------------%
 3%   This function computes the first Piola-Kirchhoff stress tensor 
 4%
 5%--------------------------------------------------------------------------
 6%--------------------------------------------------------------------------
 7
 8function   Piola       =  FirstPiolaKirchhoffStressTensorUC(Piola,ngauss,dim,F,H,DUDF,DUDH,DUDJ)
 9
10PiolaH                 =  zeros(dim,dim);
11SigmaH                 =  zeros(dim,dim);
12Fg                     =  zeros(dim,dim);
13for igauss=1:ngauss
14   %-----------------------------------------------------------------------    
15   % Variables per Gauss pointg
16   %-----------------------------------------------------------------------    
17   SigmaH              =  DUDH(:,:,igauss);
18   Fg                  =  F(:,:,igauss);
19   %-----------------------------------------------------------------------    
20   % Piola contribution associated with H
21   %-----------------------------------------------------------------------    
22    switch dim
23        case 2
24            PiolaH     =  trace(DUDH(:,:,igauss))*eye(2) - DUDH(:,:,igauss)';
25        case 3
26            PiolaH     =  [[SigmaH(2,2)*Fg(3,3) - SigmaH(2,3)*Fg(3,2) - SigmaH(3,2)*Fg(2,3) + SigmaH(3,3)*Fg(2,2)        SigmaH(2,3)*Fg(3,1) - SigmaH(2,1)*Fg(3,3) - SigmaH(3,3)*Fg(2,1) + SigmaH(3,1)*Fg(2,3)     SigmaH(2,1)*Fg(3,2) - SigmaH(2,2)*Fg(3,1) - SigmaH(3,1)*Fg(2,2) + SigmaH(3,2)*Fg(2,1)];...
27                           [SigmaH(3,2)*Fg(1,3) - SigmaH(3,3)*Fg(1,2) - SigmaH(1,2)*Fg(3,3) + SigmaH(1,3)*Fg(3,2)        SigmaH(3,3)*Fg(1,1) - SigmaH(3,1)*Fg(1,3) - SigmaH(1,3)*Fg(3,1) + SigmaH(1,1)*Fg(3,3)     SigmaH(3,1)*Fg(1,2) - SigmaH(3,2)*Fg(1,1) - SigmaH(1,1)*Fg(3,2) + SigmaH(1,2)*Fg(3,1)];...
28                           [SigmaH(1,2)*Fg(2,3) - SigmaH(1,3)*Fg(2,2) - SigmaH(2,2)*Fg(1,3) + SigmaH(2,3)*Fg(1,2)        SigmaH(1,3)*Fg(2,1) - SigmaH(1,1)*Fg(2,3) - SigmaH(2,3)*Fg(1,1) + SigmaH(2,1)*Fg(1,3)     SigmaH(1,1)*Fg(2,2) - SigmaH(1,2)*Fg(2,1) - SigmaH(2,1)*Fg(1,2) + SigmaH(2,2)*Fg(1,1)]];
29    end
30    Piola(:,:,igauss)  =  DUDF(:,:,igauss) + PiolaH + DUDJ(igauss)*H(:,:,igauss);
31end

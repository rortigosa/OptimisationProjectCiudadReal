1%--------------------------------------------------------------------------
 2%--------------------------------------------------------------------------
 3%
 4% Kinematics of the continuum
 5%
 6%--------------------------------------------------------------------------
 7%--------------------------------------------------------------------------
 8
 9function init_kinematics                =  KinematicsFunctionVolumeC(init_kinematics,dim,x_elem,DNX) 
10
11n_gauss                                 =  size(DNX,3);
12F                                       =  zeros(dim,dim);
13H                                       =  zeros(dim,dim);
14J                                       =  1;
15Dx0DX                                   =  zeros(dim,dim);
16for igauss=1:n_gauss
17    %----------------------------------------------------------------------
18    % Compute derivative of displacements (Dx0DX)
19    %----------------------------------------------------------------------
20    Dx0DX                               =  x_elem*DNX(:,:,igauss)';
21    %----------------------------------------------------------------------
22    % Compute deformation gradient tensor
23    %----------------------------------------------------------------------
24    F                                   =  Dx0DX;
25    switch dim
26        case 2
27            H  =  trace(F)*eye(2) - F';
28        case 3
29            H  =  [[F(2,2)*F(3,3) - F(2,3)*F(3,2) - F(3,2)*F(2,3) + F(3,3)*F(2,2)        F(2,3)*F(3,1) - F(2,1)*F(3,3) - F(3,3)*F(2,1) + F(3,1)*F(2,3)     F(2,1)*F(3,2) - F(2,2)*F(3,1) - F(3,1)*F(2,2) + F(3,2)*F(2,1)];...
30                [F(3,2)*F(1,3) - F(3,3)*F(1,2) - F(1,2)*F(3,3) + F(1,3)*F(3,2)        F(3,3)*F(1,1) - F(3,1)*F(1,3) - F(1,3)*F(3,1) + F(1,1)*F(3,3)     F(3,1)*F(1,2) - F(3,2)*F(1,1) - F(1,1)*F(3,2) + F(1,2)*F(3,1)];...
31                [F(1,2)*F(2,3) - F(1,3)*F(2,2) - F(2,2)*F(1,3) + F(2,3)*F(1,2)        F(1,3)*F(2,1) - F(1,1)*F(2,3) - F(2,3)*F(1,1) + F(2,1)*F(1,3)     F(1,1)*F(2,2) - F(1,2)*F(2,1) - F(2,1)*F(1,2) + F(2,2)*F(1,1)]];
32    end
33    J                          =  det(F);
34    %----------------------------------------------------------------------
35    %  Storing information.
36    %----------------------------------------------------------------------
37    init_kinematics.F(:,:,igauss)   =  F;
38    init_kinematics.H(:,:,igauss)            =  H;
39    init_kinematics.J(igauss)                =  J;
40end
